	char serial[] = "6789";

		// A
		if (name[i] < 0x41) {
			printf("Error 1\n");
			break;
		}

		// Z
		if (name[i] > 0x5A) {
			name[i] = name[i] - 0x20;
		}
	}

	// 用于serial校验的计算
	int k = 0;
	for (int i = 0; name[i] != 0; i++) {
		k += name[i];
	}
	k ^= 0x5678;

	// serical校验，将字符串数据转换为10进制数值
	int edi = 0;
	for (int i = 0, al = 0xA; serial[i] != 0; i++, al = 0xA) {
		edi = edi * al;
		edi += serial[i] - 0x30;
	}
	edi = edi ^ 0x1234;

	// k = 0x5470, edi = 0x8b1
	// 数值比较
	if (edi == k) {
		printf("succeed");
	}
	else {
		printf("Error 2\n");
	}

	return 0;
}
